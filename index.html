                resultsDiv.appendChild(loader);


                const prompt = `

                    You are an expert AI Sales Development Representative for 'Pipeline Pilot'. Your goal is to find personalized reasons to reach out to potential customers for our fintech clients. A user has provided the following website URL: ${url}.


                    **Your task:** Imagine you have visited this website.                     1.  **The Hook:** A compelling, personalized sentence that shows you've done your research.

                    chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                    const payload = { contents: chatHistory };

                    const apiKey = ""; // API key is handled by the environment

                        body: JSON.stringify(payload)
